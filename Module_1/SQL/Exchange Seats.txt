with cte1 as
(select * from seat where id=(select count(*) from seat) and id%2<>0),
cte3 as
(select id+1 as id,student from seat where id in(select id from seat where id%2<>0 and id!=(select count(*) from seat))),
cte4 as
(select id-1 as id,student from seat where id in(select id from seat where id%2=0)),
cte5 as
(select * from cte3 union select * from cte4 union select * from cte1)
select * from cte5 order by id;
